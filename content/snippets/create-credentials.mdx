## Azure


<CodeGroup>

{/* Source: /examples/sql/credential/credentials_format_azure.kisql (lines 14-17) */}
```sql Password
CREATE CREDENTIAL azure_cred
TYPE = 'azure_storage_key',
IDENTITY = 'sampleacc',
SECRET = 'foobaz123'
```

{/* Source: /examples/sql/credential/credentials_format_azure.kisql (lines 24-27) */}
```sql SAS Token
CREATE CREDENTIAL azure_cred
TYPE = 'azure_sas',
IDENTITY = 'sampleacc',
SECRET = 'sv=2015-07-08&sr=b&sig=39Up0JzHkxhUlhFEjEH9673DJxe7w6...'
```

{/* Source: /examples/sql/credential/credentials_format_azure.kisql (lines 34-42) */}
```sql Active Directory
CREATE CREDENTIAL azure_cred
TYPE = 'azure_ad',
IDENTITY = 'jdoe',
SECRET = 'foobaz123'
WITH OPTIONS
(
    STORAGE ACCOUNT NAME = 'sampleacc',
    TENANT ID = 'x0xxx10-00x0-0x01-0xxx-x0x0x01xx100'
)
```

</CodeGroup>

## Google Cloud Storage


<CodeGroup>

{/* Source: /examples/sql/credential/credentials_format_gcs.kisql (lines 36-37 14-15) */}
```sql User ID & Key
CREATE CREDENTIAL gcs_cred
TYPE = 'gcs_service_account_id',
IDENTITY = 'auser@auser.iam.gserviceaccount.com',
SECRET = '-----BEGIN PRIVATE KEY-----\nABCDEFG=\n-----END PRIVATE KEY-----\n'
```

{/* Source: /examples/sql/credential/credentials_format_gcs.kisql (lines 47-50 17-29 64) */}
```sql JSON Key
CREATE CREDENTIAL gcs_cred
TYPE = 'gcs_service_account_keys'
WITH OPTIONS
(
	GCS_SERVICE_ACCOUNT_KEYS = '
	{
		"type": "service_account",
		"project_id": "auser",
		"private_key_id": "abcdef1234567890",
		"private_key": "-----BEGIN PRIVATE KEY-----\nABCDEFG=\n-----END PRIVATE KEY-----\n",
		"client_email": "auser@auser.iam.gserviceaccount.com",
		"client_id": "1234567890",
		"auth_uri": "https://accounts.google.com/o/oauth2/auth",
		"token_uri": "https://oauth2.googleapis.com/token",
		"auth_provider_x509_cert_url": "https://www.googleapis.com/oauth2/v1/certs",
		"client_x509_cert_url": "https://www.googleapis.com/robot/v1/metadata/x509/auser%40auser.iam.gserviceaccount.com"
	}'
)
```

</CodeGroup>

## HDFS


<CodeGroup>

{/* Source: /examples/sql/credential/credentials_format_hdfs.kisql (lines 14-17) */}
```sql Password
CREATE CREDENTIAL hdfs_cred
TYPE = 'hdfs',
IDENTITY = 'jdoe',
SECRET = 'foobaz123'
```

{/* Source: /examples/sql/credential/credentials_format_hdfs.kisql (lines 37-43) */}
```sql Kerberos Keytab
CREATE CREDENTIAL hdfs_cred
TYPE = 'hdfs',
IDENTITY = 'jdoe'
WITH OPTIONS
(
    KERBEROS KEYTAB = 'kifs://<keytab file path>'
)
```

{/* Source: /examples/sql/credential/credentials_format_hdfs.kisql (lines 24-30) */}
```sql Kerberos Token
CREATE CREDENTIAL hdfs_cred
TYPE = 'hdfs',
IDENTITY = 'jdoe'
WITH OPTIONS
(
    USE KERBEROS = 'true'
)
```

</CodeGroup>

## Kafka (Apache)


<CodeGroup>

{/* Source: /examples/sql/credential/credentials_format_kafka.kisql (lines 24-30) */}
```sql SSL with CA Cert
CREATE CREDENTIAL kafka_cred
TYPE = 'kafka'
WITH OPTIONS
(
    'security.protocol' = 'SSL',
    'ssl.ca.location' = 'kifs://ssl/ca-bundle.crt'
)
```

{/* Source: /examples/sql/credential/credentials_format_kafka.kisql (lines 37-44) */}
```sql SSL with CA Cert/Client Auth
CREATE CREDENTIAL kafka_cred
TYPE = 'kafka'
WITH OPTIONS
(
    'security.protocol' = 'SSL',
    'ssl.ca.location' = 'kifs://ssl/ca-bundle.crt',
    'ssl.certificate.location' = 'kifs://ssl/client.pem'
)
```

{/* Source: /examples/sql/credential/credentials_format_kafka.kisql (lines 51-60) */}
```sql SSL with Encryption
CREATE CREDENTIAL kafka_cred
TYPE = 'kafka'
WITH OPTIONS
(
    'security.protocol' = 'SSL',
    'ssl.ca.location' = 'kifs://ssl/ca-bundle.crt',
    'ssl.certificate.location' = 'kifs://ssl/client.pem',
    'ssl.key.location' = 'kifs://ssl/client.key',
    'ssl.key.password' = 'foobaz123'
)
```

{/* Source: /examples/sql/credential/credentials_format_kafka.kisql (lines 67-75) */}
```sql SASL
CREATE CREDENTIAL kafka_cred
TYPE = 'kafka'
WITH OPTIONS
(
    'security.protocol' = 'SASL_SSL',
    'sasl.mechanism' = 'PLAIN',
    'sasl.username' = 'jdoe',
    'sasl.password' = 'foobaz123'
)
```

{/* Source: /examples/sql/credential/credentials_format_kafka.kisql (lines 82-91) */}
```sql Kerberos
CREATE CREDENTIAL kafka_cred
TYPE = 'kafka'
WITH OPTIONS
(
    'security.protocol' = 'SASL_PLAINTEXT',
    'sasl.mechanism' = 'GSSAPI',
    'sasl.kerberos.service.name' = 'kafka',
    'sasl.kerberos.keytab' = 'kifs://security/jdoe.keytab',
    'sasl.kerberos.principal' = 'jdoe@example.com'
)
```

{/* Source: /examples/sql/credential/credentials_format_kafka.kisql (lines 98-111) */}
```sql Kerberos SSL
CREATE CREDENTIAL kafka_cred
TYPE = 'kafka'
WITH OPTIONS
(
    'security.protocol' = 'SASL_SSL',
    'sasl.mechanism' = 'GSSAPI',
    'sasl.kerberos.service.name' = 'kafka',
    'sasl.kerberos.keytab' = 'kifs://security/jdoe.keytab',
    'sasl.kerberos.principal' = 'jdoe@example.com',
    'ssl.ca.location' = 'kifs://ssl/ca-bundle.crt',
    'ssl.certificate.location' = 'kifs://ssl/client.pem',
    'ssl.key.location' = 'kifs://ssl/client.key',
    'ssl.key.password' = 'foobaz123'
)
```

</CodeGroup>

## S3 (Amazon)


<CodeGroup>

{/* Source: /examples/sql/credential/credentials_format_s3.kisql (lines 14-17) */}
```sql S3 Access Key
CREATE CREDENTIAL s3_cred
TYPE = 'aws_access_key',
IDENTITY = 'AKIAIOSFODNN7EXAMPLE',
SECRET = 'wJalrXUtnFEMI/K7MDENG/bPxRfiCYEXAMPLEKEY'
```

{/* Source: /examples/sql/credential/credentials_format_s3.kisql (lines 24-31) */}
```sql IAM Role
CREATE CREDENTIAL s3_cred
TYPE = 'aws_iam_role',
IDENTITY = 'AKIAIOSFODNN7EXAMPLE',
SECRET = 'wJalrXUtnFEMI/K7MDENG/bPxRfiCYEXAMPLEKEY'
WITH OPTIONS
(
    S3_AWS_ROLE_ARN = 'arn:aws:iam::123456789012:user/JohnDoe'
)
```

</CodeGroup>
